Private Sub btnInsert_Click()
If DBform.btnInsert.Caption = "Insert" Then
  Dim mensaje As VbMsgBoxResult
  Dim valid As String
    mensaje = MsgBox("Do you want to insert the data?", vbYesNo + vbInformation, "Confirmation")
    If mensaje = vbNo Then Exit Sub
    valid = validateRecords
    If valid = "" Then
    Call insert
    Call clear
    Else
    MsgBox valid, vbExclamation
    End If
Else
    Dim mensaje2 As VbMsgBoxResult
  Dim valid2 As String
    mensaje2 = MsgBox("Do you want to update the data?", vbYesNo + vbInformation, "Confirmation")
    If mensaje2 = vbNo Then Exit Sub
    valid2 = validateRecords
    If valid2 = "" Then
    Call update
    Call cancel
    Call searchValue
    Else
    MsgBox valid2, vbExclamation
    End If
End If

End Sub

Private Sub btnClear_Click()
If DBform.btnClear.Caption = "Clear" Then
Call clear
Else
Call cancel
Call searchValue
End If

End Sub
